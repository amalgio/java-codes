3        int[] arr = {5, 1, 4, 2, 8};
4        for (int i = 0; i < arr.length - 1; i++) {
5            for (int j = 0; j < arr.length - 1 - i; j++) {
6                if (arr[j] > arr[j + 1]) {
7                    int temp = arr[j];
8                    arr[j] = arr[j + 1];

10                }
11            }
12        }

1class InsertionSort {
2    public static void main(String[] args) {
3        int[] arr = {5, 3, 1, 4, 2};
4        for (int i = 1; i < arr.length; i++) {
5            int key = arr[i];
6            int j = i - 1;
7            while (j >= 0 && arr[j] > key) {
8                arr[j + 1] = arr[j];
9                j--;
10            }
11            arr[j + 1] = key;
12        }
13        for (int i = 0; i < arr.length; i++) {
14            System.out.print(arr[i] + " ");
15        }
16    }
17}

EX 12 Merge Sort Java
▼
MergeSort.java 
1public class MergeSort {
2    public static void main(String[] args) {
3        int[] arr = {3,1,2,6,5};
4        int n = arr.length;
5        for(int currsize = 1; currsize <= n-1; currsize = 2*currsize) {
6            for(int leftstart=0; leftstart<n-1; leftstart+=2*currsize) {
7                int mid = Math.min(leftstart+currsize-1, n-1);
8                int rightend = Math.min(leftstart+2*currsize-1, n-1);
9                int n1 = mid-leftstart+1, n2 = rightend-mid;
10                int[] l = new int[n1], r = new int[n2];
11                for(int i=0; i<n1; i++) l[i]=arr[leftstart+i];
12                for(int j=0; j<n2; j++) r[j]=arr[mid+1+j];
13                int i=0, j=0, k=leftstart;
14                while(i<n1 && j<n2) {
15                    if(l[i]<=r[j]) arr[k++]=l[i++];
16                    else arr[k++]=r[j++];
17                }
18                while(i<n1) arr[k++]=l[i++];
19                while(j<n2) arr[k++]=r[j++];
20            }
21        }
22        System.out.println("Sorted Array:");
23        for(int val : arr) System.out.println(val+"");
24    }
25}
Output
Sorted Array: 1 2 3 5 6

QuickSort.java 
1class QuickSort {
2    public static void main(String[] args) {
3        int[] arr = {2, 3, 1, 4, 5, 6};
4        int n = arr.length;
5        int[] stack = new int[2 * n];
6        int top = -1;
7        stack[++top] = 0; stack[++top] = n - 1;
8        while (top >= 0) {
9            int h = stack[top--], l = stack[top--];
10            int pivot = arr[h], i = l - 1;
11            for (int j = l; j < h; j++) {
12                if (arr[j] <= pivot) { i++; int t=arr[i]; arr[i]=arr[j]; arr[j]=t; }
13            }
14            int t=arr[i+1]; arr[i+1]=arr[h]; arr[h]=t;
15            int p = i + 1;
16            if (p - 1 > l) { stack[++top]=l; stack[++top]=p-1; }
17            if (p + 1 < h) { stack[++top]=p+1; stack[++top]=h; }
18        }
19        System.out.println("Sorted:");
20        for (int val : arr) System.out.println(val);
21    }
22}
Output
Sorted: 1 2 3 4 5 6

1abstract class Shape {

3    abstract void printArea();
4}
5class Rectangle extends Shape {

14        System.out.println("Area of Triangle: " + (a * b) / 2.0);
15    }
16}
17class Circle extends Shape {
18    Circle(int radius) { a = radius; }

20        System.out.println("Area of Circle: " + Math.PI * a * a);

24    public static void main(String[] args) {
25        Shape r = new Rectangle(10, 5);
26        r.printArea();

31    }

EX 16 Bank Account — Exception Handling Java
▼
ExceptionHandling.java 
1import java.util.Scanner;
2
3class InsufficientBalanceException extends Exception {
4    InsufficientBalanceException(String message) { super(message); }
5}
6class BankAccount {
7    private double balance;
8    BankAccount(double balance) { this.balance = balance; }
9    void deposit(double amount) throws Exception {
10        if (amount <= 0) throw new IllegalArgumentException("Deposit amount must be positive!");
11        balance += amount;
12        System.out.println("Deposit successful! Current balance: " + balance);
13    }
14    void withdraw(double amount) throws Exception {
15        if (amount <= 0) throw new IllegalArgumentException("Withdrawal amount must be positive!");
16        if (amount > balance) throw new InsufficientBalanceException("Not enough balance!");
17        balance -= amount;
18        System.out.println("Withdrawal successful! Current balance: " + balance);
19    }
20    void checkBalance() { System.out.println("Current balance: " + balance); }
21}
22public class ExceptionHandling {
23    public static void main(String[] args) {
24        Scanner input = new Scanner(System.in);
25        BankAccount acc = new BankAccount(10000);
26        while (true) {
27            System.out.println("\n1.Deposit 2.Withdraw 3.Check Balance 4.Exit");
28            System.out.print("Enter choice: ");
29            try {
30                int choice = input.nextInt();
31                if (choice == 1) { System.out.print("Enter amount: "); acc.deposit(input.nextDouble()); }
32                else if (choice == 2) { System.out.print("Enter amount: "); acc.withdraw(input.nextDouble()); }
33                else if (choice == 3) { acc.checkBalance(); }
34                else if (choice == 4) { System.out.println("Thank you!"); break; }
35                else System.out.println("Invalid choice");
36            } catch (InsufficientBalanceException e) { System.out.println(e.getMessage()); }
37              catch (IllegalArgumentException e) { System.out.println(e.getMessage()); }
38              catch (Exception e) { System.out.println("Invalid input"); input.nextLine(); }
39        }
40        input.close();
41    }
42}

1public class BinarySearch {
2    public static int binarySearch(int[] arr, int target) {
3        int low = 0, high = arr.length - 1;
4        while (low <= high) {
5            int mid = (low + high) / 2;
6            if (arr[mid] == target) return mid;
7            else if (arr[mid] < target) low = mid + 1;
8            else high = mid - 1;
9        }
10        return -1;
11    }
12    public static void main(String[] args) {
13        int[] arr = {7, 14, 23, 35, 48, 56, 62, 78};
14        int target = 48;
15        int result = binarySearch(arr, target);

17        else System.out.println("Not found!");
18    }
19}
